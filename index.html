function drawMatrix() { ctx.fillStyle = "rgba(0, 0, 0, 0.05)"; ctx.fillRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = "#00f2ff"; ctx.font = fontSize + "px monospace"; for (let i = 0; i < drops.length; i++) { const char = chars.charAt(Math.floor(Math.random() * chars.length)); ctx.fillText(char, i * fontSize, drops[i] * fontSize); if (drops[i] * fontSize > canvas.height && Math.random() > 0.975) { drops[i] = 0; } drops[i]++; } } setInterval(drawMatrix, 30);
